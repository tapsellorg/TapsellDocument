ШіШ§Ш®ШӘ ШӘШЁЩ„ЫҢШәЪҜШ§ЩҮ

Ш§ШЁШӘШҜШ§ Ш§ШІ ЩҫЩҶЩ„ ШӘЩҫШіЩ„ ЫҢЪ© ШӘШЁЩ„ЫҢШәЪҜШ§ЩҮ (zone) ЩҮЩ…ШіШ§ЩҶ ШЁШіШ§ШІЫҢШҜ ЩҲ zoneId ШұШ§ ШІЩ…Ш§ЩҶ ШҜШұШ®ЩҲШ§ШіШӘ ЩҲ ЩҶЩ…Ш§ЫҢШҙ ШӘШЁЩ„ЫҢШә Ш§ШіШӘЩҒШ§ШҜЩҮ Ъ©ЩҶЫҢШҜ.

ШҜШұШ®ЩҲШ§ШіШӘ ШӘШЁЩ„ЫҢШә

ШЁШ§ Ъ©Щ…Ъ© Щ…ШӘШҜ TapsellPlus.RequestNativeBannerAd ЩҲ ШЁЩҮ ШұЩҲШҙ ШІЫҢШұ ШҜШұШ®ЩҲШ§ШіШӘ ШӘШЁЩ„ЫҢШә ШЁШҜЩҮЫҢШҜ.

TapsellPlus.RequestNativeBannerAd(ZoneID,

			tapsellPlusAdModel => {
				Debug.Log ("On Response " + tapsellPlusAdModel.responseId);
				_responseId = tapsellPlusAdModel.responseId;
			},
			error => {
				Debug.Log ("Error " + error.message);
			}
		);

Ш§ЪҜШұ ШӘЩ…Ш§ЫҢЩ„ ШҜШ§ШұЫҢШҜ ШҜШұ Ъ©Ш§Щ„ШЁЪ© error Щ…Ш¬ШҜШҜШ§ ШҜШұШ®ЩҲШ§ШіШӘ ШӘШЁЩ„ЫҢШә Ъ©ЩҶЫҢШҜШҢ ШӯШӘЩ…Ш§ Ш§ЫҢЩҶ Ъ©Ш§Шұ ШұШ§ ШЁЩҮ Ъ©Щ…Ъ© Щ…ШӘШәЫҢШұЫҢ ШЁЩҮ Ш№ЩҶЩҲШ§ЩҶ ШҙЩ…Ш§ШұЩҶШҜЩҮ Ш§ЩҶШ¬Ш§Щ… ШҜЩҮЫҢШҜ. ШІЫҢШұШ§ ШЁЩҮ Ъ©Щ…Ъ© ШўЩҶ Щ…ШӘШәЫҢШұ Щ…ЫҢвҖҢШӘЩҲШ§ЩҶЫҢШҜ Щ…ШӯШҜЩҲШҜЫҢШӘ ШӘШ№ШҜШ§ШҜ ШҜЩҒШ№Ш§ШӘ ШұШ§ ШЁШұШ§ЫҢ ШҜШұШ®ЩҲШ§ШіШӘ Щ„ШӯШ§Шё Ъ©ЩҶЫҢШҜ. ШЁЩҮ Ш№ЩҶЩҲШ§ЩҶ Щ…Ш«Ш§Щ„ ЩҲЩӮШӘЫҢ Ш§ЫҢЩҶ Ш¬Ш§ЫҢЪҜШ§ЩҮ ШӘШЁЩ„ЫҢШәШ§ШӘЫҢ ШұШ§ Ш§ШІ ЩҫЩҶЩ„ ШәЫҢШұЩҒШ№Ш§Щ„ ЩҶЩ…ЩҲШҜЫҢШҜШҢ Ш§ЪҜШұ ШЁШҜЩҲЩҶ Щ…ШӯШҜЩҲШҜ Ъ©ШұШҜЩҶ ШҜЩҒШ№Ш§ШӘШҢ ЩҮШұ ШЁШ§Шұ ШҜШұ Ъ©Ш§Щ„ШЁЪ© error Щ…Ш¬ШҜШҜШ§ ШҜШұШ®ЩҲШ§ШіШӘ ШӘШЁЩ„ЫҢШә ШҜЩҮЫҢШҜШҢ ШЁШұЩҶШ§Щ…ЩҮвҖҢШӘШ§ЩҶ ШҜШұ ЫҢЪ© ШӯЩ„ЩӮЩҮвҖҢЫҢ ШЁЫҢвҖҢЩҶЩҮШ§ЫҢШӘ Щ…ЫҢвҖҢШ§ЩҒШӘШҜ ЩҲ Ш№Щ…Щ„Ъ©ШұШҜ ШўЩҶ Щ…Ш®ШӘЩ„ Щ…ЫҢвҖҢШҙЩҲШҜ.

ЩҶЩ…Ш§ЫҢШҙ ШӘШЁЩ„ЫҢШә

ШЁШ№ШҜ Ш§ШІ Ш§Ш¬ШұШ§ЫҢ Щ…ШӘШҜ response ЩҲ ШҜШұЫҢШ§ЩҒШӘ ЩҫШ§ШұШ§Щ…ШӘШұ responseId ШӘШЁЩ„ЫҢШә ШўЩ…Ш§ШҜЩҮ ЩҶЩ…Ш§ЫҢШҙ Ш§ШіШӘ ЩҲ Щ…ЫҢвҖҢШӘЩҲШ§ЩҶЫҢШҜ Щ…Ш·Ш§ШЁЩӮ ШұЩҲШҙ ШІЫҢШұ ШўЩҶ ШұШ§ ЩҶЩ…Ш§ЫҢШҙ ШҜЩҮЫҢШҜ.

public void Show () {
  // this: refers to MonoBehaviour
  TapsellPlus.ShowNativeBannerAd(_responseId, this,

			tapsellPlusNativeBannerAd => {
				Debug.Log ("onOpenAd " + tapsellPlusNativeBannerAd.zoneId);
				adHeadline.text = ArabicSupport.ArabicFixer.Fix(tapsellPlusNativeBannerAd.title);
				adCallToAction.text = ArabicSupport.ArabicFixer.Fix(tapsellPlusNativeBannerAd.callToActionText);
				adBody.text = ArabicSupport.ArabicFixer.Fix(tapsellPlusNativeBannerAd.description);
				adImage.texture = tapsellPlusNativeBannerAd.landscapeBannerImage;
        
				tapsellPlusNativeBannerAd.RegisterImageGameObject(adImage.gameObject);
				tapsellPlusNativeBannerAd.RegisterHeadlineTextGameObject(adHeadline.gameObject);
				tapsellPlusNativeBannerAd.RegisterCallToActionGameObject(adCallToAction.gameObject);
				tapsellPlusNativeBannerAd.RegisterBodyTextGameObject(adBody.gameObject);
			},
			error => {
				Debug.Log ("onError " + error.errorMessage);
			}
		);
}

descriptionstringШӘЩҲШ¶ЫҢШӯШ§ШӘ

callToActionTextstringЩ…ШӘЩҶ ШҜЪ©Щ…ЩҮ Ъ©Щ„ЫҢЪ©

ШЁЩҮ Ш№ЩҶЩҲШ§ЩҶ Щ…Ш«Ш§Щ„ Ш§ЪҜШұ ЫҢЪ© GameObject Ш§ШІ ЩҶЩҲШ№ RawImage ШЁЩҮ ЩҶШ§Щ… adImage ШҜШұ Ш§Ш®ШӘЫҢШ§Шұ ШҜШ§ШұЫҢШҜШҢ ШЁЩҮ Ъ©Щ…Ъ© ШӘЪ©ЩҮ Ъ©ШҜ ШІЫҢШұ Щ…ЫҢвҖҢШӘЩҲШ§ЩҶЫҢШҜ ШӘШөЩҲЫҢШұ ШӘШЁЩ„ЫҢШә ШұШ§ ШҜШұ ШўЩҶ ЩҶЩ…Ш§ЫҢШҙ ШҜЩҮЫҢШҜ:

adImage.texture = tapsellPlusNativeBannerAd.landscapeBannerImage;
...

ШЁШ§ШІ Ъ©ШұШҜЩҶ ШӘШЁЩ„ЫҢШә

ШЁШұШ§ЫҢ ШЁШ§ШІ Ъ©ШұШҜЩҶ ШӘШЁЩ„ЫҢШә Щ…ЫҢвҖҢШЁШ§ЫҢШҜ Ш§ШЁШӘШҜШ§ ШЁЩҮ ЩҮШұ Ъ©ШҜШ§Щ… Ш§ШІ GameObjectЩҮШ§ЫҢЫҢ Ъ©ЩҮ ШҜШұ Щ…ШұШӯЩ„ЩҮвҖҢЫҢ ЩҫЫҢШҙ Ш§ЫҢШ¬Ш§ШҜ Ъ©ШұШҜЩҮвҖҢШ§ЫҢШҜШҢ ЫҢЪ© Component Ш§ШІ Ш¬ЩҶШі Box Collider Ш§Ш¶Ш§ЩҒЩҮ Ъ©ЩҶЫҢШҜ ЩҲ ШіЩҫШі Game Object ШұШ§ Ш§ШІ Ш·ШұЫҢЩӮ Щ…ШӘШҜЩҮШ§ЫҢ ШІЫҢШұ ШЁЩҮ ШӘЩҫШіЩ„вҖҢЩҫЩ„Ш§Ші Щ…Ш№ШұЩҒЫҢ Ъ©ЩҶЫҢШҜ.

ШҜЩӮШӘ Ъ©ЩҶЫҢШҜ Ъ©ЩҮ ШЁШұШ§ЫҢ ЩҮШұ Ъ©ШҜШ§Щ… Ш§ШІ Game ObjectЩҮШ§ ШӘЫҢЪ© Raycast Target ШұШ§ ШІШҜЩҮ ШЁШ§ШҙЫҢШҜ ШЁЩҮ Ш¬ШІ Button Ъ©ЩҮ ШӘЫҢЪ© Raycast Target ШұШ§ Ш§ШІ Ъ©Ш§Щ…ЩҫЩҲЩҶЩҶШӘ Image ШўЩҶ ШЁШ§ЫҢШҜ ШЁШұШҜШ§ШұЫҢШҜ.
